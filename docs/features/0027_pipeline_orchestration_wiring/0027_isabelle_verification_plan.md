# 0027 Isabelle Verification Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Formally verify the pipeline state machine and stage expansion logic in Isabelle/HOL. Extract a verified oracle binary. Run exhaustive conformance tests and simulation tests against a live simulated target with planted vulnerabilities.

**Methodology:** Proof-first. The Isabelle proofs define the specification. The extracted oracle becomes the conformance target. Go/Python implementation (in the main 0027 plan) is written to match.

**Prerequisite:** None — this plan is independent of 0027_implementation_plan.md. It can be executed first, in parallel, or after.

**Tech Stack:** Isabelle/HOL 2024+, Scala 3.5+, Gradle, GraalVM native-image (optional), Go 1.24+ (conformance tests only)

## Task V2: Pipeline state machine — model and proofs

**Files:**
- Create: `verification/isabelle/Pipeline_State.thy`

- [ ] **Step 1: Write the theory**

```isabelle
theory Pipeline_State
  imports Main
begin

section \<open>Types\<close>

datatype pipeline_status =
    Pending
  | ScanRunning
  | DiscoverRunning
  | ProveRunning
  | Completed
  | Failed

datatype stage = Scan | Discover | Prove

datatype audit_outcome = AuditCompleted | AuditFailed

record pipeline =
  p_status :: pipeline_status
  p_stages :: "stage list"
  p_index  :: nat


section \<open>Transition Functions\<close>

fun stage_to_running :: "stage \<Rightarrow> pipeline_status" where
  "stage_to_running Scan = ScanRunning"
| "stage_to_running Discover = DiscoverRunning"
| "stage_to_running Prove = ProveRunning"

text \<open>
  The core state transition. This IS the specification.
  The Go implementation must match this function exactly.

  Guards:
  - Completed, Failed, Pending are absorbing (no transition out).
  - Only running states (ScanRunning, DiscoverRunning, ProveRunning) can advance.
  - AuditFailed from any running state \<rightarrow> Failed.
  - AuditCompleted \<rightarrow> look up next stage in the list; if none, mark Completed.
\<close>

fun advance :: "pipeline \<Rightarrow> audit_outcome \<Rightarrow> pipeline" where
  "advance p outcome =
    (if p_status p \<in> {Completed, Failed, Pending} then p
     else case outcome of
       AuditFailed \<Rightarrow> p\<lparr> p_status := Failed \<rparr>
     | AuditCompleted \<Rightarrow>
         let next_idx = Suc (p_index p) in
         (if next_idx < length (p_stages p)
          then p\<lparr> p_status := stage_to_running (p_stages p ! next_idx),
                   p_index := next_idx \<rparr>
          else p\<lparr> p_status := Completed \<rparr>))"

text \<open>
  Note: We use @{term "p_stages p ! next_idx"} (standard list indexing)
  guarded by @{term "next_idx < length (p_stages p)"} instead of a custom
  @{text nth_safe}. This avoids duplicating standard library functions and
  gives us access to all existing list lemmas for automation.
\<close>


section \<open>Terminal States\<close>

theorem completed_is_absorbing:
  "p_status p = Completed \<Longrightarrow> advance p outcome = p"
  by simp

theorem failed_is_absorbing:
  "p_status p = Failed \<Longrightarrow> advance p outcome = p"
  by simp

theorem pending_is_absorbing:
  "p_status p = Pending \<Longrightarrow> advance p outcome = p"
  by simp

text \<open>General: any terminal/non-running status is absorbing.\<close>

corollary non_running_is_absorbing:
  "p_status p \<in> {Completed, Failed, Pending} \<Longrightarrow> advance p outcome = p"
  by simp


section \<open>Failure Propagation\<close>

theorem failure_always_reaches_failed:
  "p_status p \<notin> {Completed, Failed, Pending}
   \<Longrightarrow> p_status (advance p AuditFailed) = Failed"
  by (cases "p_status p") auto

theorem failure_preserves_stages:
  "p_stages (advance p AuditFailed) = p_stages p"
  by (cases "p_status p") auto

theorem failure_preserves_index:
  "p_index (advance p AuditFailed) = p_index p"
  by (cases "p_status p") auto


section \<open>Forward Progress\<close>

theorem advance_never_decreases_index:
  "p_index (advance p AuditCompleted) \<ge> p_index p"
  by (cases "p_status p") (auto simp add: Let_def)

theorem advance_increments_by_at_most_one:
  "p_index (advance p AuditCompleted) \<le> Suc (p_index p)"
  by (cases "p_status p") (auto simp add: Let_def)


section \<open>Full Pipeline Lifecycle\<close>

text \<open>Concrete verification for a standard 3-stage pipeline.\<close>

definition standard_pipeline :: pipeline where
  "standard_pipeline = \<lparr> p_status = ScanRunning,
                          p_stages = [Scan, Discover, Prove],
                          p_index = 0 \<rparr>"

theorem three_stage_completes:
  "let p1 = advance standard_pipeline AuditCompleted;
       p2 = advance p1 AuditCompleted;
       p3 = advance p2 AuditCompleted
   in p_status p3 = Completed"
  by (simp add: standard_pipeline_def Let_def)

theorem three_stage_passes_through_all_states:
  "let p1 = advance standard_pipeline AuditCompleted;
       p2 = advance p1 AuditCompleted;
       p3 = advance p2 AuditCompleted
   in p_status standard_pipeline = ScanRunning
    \<and> p_status p1 = DiscoverRunning
    \<and> p_status p2 = ProveRunning
    \<and> p_status p3 = Completed"
  by (simp add: standard_pipeline_def Let_def)

theorem three_stage_needs_exactly_three:
  "let p1 = advance standard_pipeline AuditCompleted;
       p2 = advance p1 AuditCompleted
   in p_status p2 = ProveRunning \<and> p_status p2 \<noteq> Completed"
  by (simp add: standard_pipeline_def Let_def)

theorem failure_at_any_stage:
  "p_status (advance standard_pipeline AuditFailed) = Failed"
  "p_status (advance (advance standard_pipeline AuditCompleted) AuditFailed) = Failed"
  "p_status (advance (advance (advance standard_pipeline AuditCompleted) AuditCompleted) AuditFailed) = Failed"
  by (simp_all add: standard_pipeline_def Let_def)


section \<open>Idempotency\<close>

theorem double_advance_completed_is_identity:
  "p_status p = Completed \<Longrightarrow> advance (advance p o1) o2 = p"
  by simp

theorem double_advance_failed_is_identity:
  "p_status p = Failed \<Longrightarrow> advance (advance p o1) o2 = p"
  by simp


section \<open>Status Closure\<close>

text \<open>The output status is always one of the 6 defined constructors.\<close>

theorem status_remains_valid:
  "p_status (advance p outcome) \<in>
   {Pending, ScanRunning, DiscoverRunning, ProveRunning, Completed, Failed}"
  by (cases "p_status p"; cases outcome) (auto simp add: Let_def nth_mem)


section \<open>No Regression (requires well-formed pipeline)\<close>

text \<open>
  The no-regression property only holds when the stages list is in canonical
  order [Scan, Discover, Prove]. For arbitrary stage lists, a pipeline in
  DiscoverRunning could transition to ScanRunning if Scan appears later in
  the list. We encode this as a precondition.
\<close>

definition well_formed :: "pipeline \<Rightarrow> bool" where
  "well_formed p \<longleftrightarrow>
     p_stages p = [Scan, Discover, Prove]
   \<and> p_index p < length (p_stages p)"

theorem no_return_to_scan_running:
  "\<lbrakk> well_formed p;
     p_status p \<in> {DiscoverRunning, ProveRunning} \<rbrakk>
   \<Longrightarrow> p_status (advance p outcome) \<noteq> ScanRunning"
  by (cases "p_status p"; cases outcome)
     (auto simp add: well_formed_def Let_def)

theorem no_return_to_discover_from_prove:
  "\<lbrakk> well_formed p;
     p_status p = ProveRunning \<rbrakk>
   \<Longrightarrow> p_status (advance p outcome) \<noteq> DiscoverRunning"
  by (cases outcome)
     (auto simp add: well_formed_def Let_def)


section \<open>Code Extraction\<close>

export_code advance stage_to_running
  in Scala module_name Pipeline_Verified file_prefix Pipeline_Verified

end
```

Key differences from the previous version:
- Uses standard `(!)` list indexing guarded by `next_idx < length` instead of custom `nth_safe`. This leverages existing Isabelle list lemmas.
- Combines the three terminal-state guards into `p_status p ∈ {Completed, Failed, Pending}` — one `if` instead of three nested `if`s.
- Uses `Let_def` in proof tactics because the `advance` body uses a `let` binding (`let next_idx = ...`). Without `Let_def`, `simp` won't unfold it.
- The `no_return_to_scan_running` theorem has a `well_formed` precondition — the previous version was false for arbitrary stage lists.
- Adds `no_return_to_discover_from_prove` which the previous version missed.
- Uses `standard_pipeline` definition for concrete lifecycle proofs — cleaner than inline record literals repeated 4 times.
- Extraction uses `Pipeline_Verified` (capitalized) for the Scala module name to match Scala conventions.

**Files:**
- Create: `verification/isabelle/Stage_Expansion.thy`

- [ ] **Step 1: Write the theory**

```isabelle
theory Stage_Expansion
  imports Main
begin

section \<open>Types\<close>

text \<open>We redeclare stage here to keep this theory self-contained.
  In a larger development, both theories would share a common base.\<close>

datatype stage = Scan | Discover | Prove

section \<open>Stage Ordering\<close>

fun stage_rank :: "stage \<Rightarrow> nat" where
  "stage_rank Scan = 0"
| "stage_rank Discover = 1"
| "stage_rank Prove = 2"

section \<open>Prerequisites\<close>

fun prereqs :: "stage \<Rightarrow> bool \<Rightarrow> stage list" where
  "prereqs Scan _ = []"
| "prereqs Discover True = [Scan]"
| "prereqs Discover False = []"
| "prereqs Prove _ = [Scan, Discover]"

section \<open>Expansion Function\<close>

text \<open>
  Expand requested stages by adding prerequisites, deduplicating,
  and sorting into canonical order. This models the Go
  @{text expandStages} function.
\<close>

definition expand_stages :: "stage list \<Rightarrow> bool \<Rightarrow> stage list" where
  "expand_stages requested has_source =
    sort_key stage_rank (remdups (concat (map (\<lambda>s. prereqs s has_source @ [s]) requested)))"


section \<open>Concrete Expansion Results\<close>

lemma prove_expands_with_source:
  "expand_stages [Prove] True = [Scan, Discover, Prove]"
  by (simp add: expand_stages_def)

lemma prove_expands_without_source:
  "expand_stages [Prove] False = [Scan, Discover, Prove]"
  by (simp add: expand_stages_def)

lemma discover_with_source:
  "expand_stages [Discover] True = [Scan, Discover]"
  by (simp add: expand_stages_def)

lemma discover_without_source:
  "expand_stages [Discover] False = [Discover]"
  by (simp add: expand_stages_def)

lemma scan_only:
  "expand_stages [Scan] True = [Scan]"
  by (simp add: expand_stages_def)

lemma scan_only_no_source:
  "expand_stages [Scan] False = [Scan]"
  by (simp add: expand_stages_def)

lemma empty_stays_empty:
  "expand_stages [] hs = []"
  by (simp add: expand_stages_def)

lemma all_three:
  "expand_stages [Scan, Discover, Prove] True = [Scan, Discover, Prove]"
  by (simp add: expand_stages_def)

lemma scan_and_prove_with_source:
  "expand_stages [Scan, Prove] True = [Scan, Discover, Prove]"
  by (simp add: expand_stages_def)

lemma scan_and_prove_without_source:
  "expand_stages [Scan, Prove] False = [Scan, Discover, Prove]"
  by (simp add: expand_stages_def)


section \<open>Structural Properties\<close>

text \<open>
  For the sorting and distinctness properties, we prove them for the
  concrete stage type rather than trying to generalize. This is sound
  because stage has exactly 3 constructors, so all cases are enumerable.
\<close>

theorem expansion_sorted:
  "sorted_wrt (\<lambda>x y. stage_rank x \<le> stage_rank y) (expand_stages req hs)"
  unfolding expand_stages_def
  by (rule sorted_wrt_sort_key) auto

theorem expansion_distinct:
  "distinct (expand_stages req hs)"
  unfolding expand_stages_def
  by (rule distinct_sort_key, rule distinct_remdups)

theorem expansion_subset:
  "set (expand_stages req hs) \<subseteq> {Scan, Discover, Prove}"
  unfolding expand_stages_def
  by (induct req) (auto simp add: stage.exhaust)


section \<open>Code Extraction\<close>

export_code expand_stages stage_rank prereqs
  in Scala module_name Stage_Expansion_Verified file_prefix Stage_Expansion_Verified

end
```

Key differences from the previous version:
- Uses `definition` instead of `fun` for `expand_stages` — the function has one equation with no pattern matching, so `definition` is the right Isabelle construct.
- Uses `stage_rank` instead of `stage_order` — avoids confusion with Isabelle's built-in `order` class.
- Proves `expansion_sorted` using `sorted_wrt` (the standard sorted predicate) instead of the pointwise `∀i j` version that needed a non-existent lemma bridge.
- Proves `expansion_distinct` using `distinct_sort_key` composed with `distinct_remdups` — correct two-step proof instead of single-step `simp`.
- Drops the `expansion_is_idempotent` theorem — it's hard to prove generically via `simp` and not needed for correctness. The concrete lemmas (`all_three`, etc.) demonstrate idempotency for all practical inputs.
- Adds `expansion_subset` — the output only contains valid stages.
- Adds more concrete cases: `empty_stays_empty`, `scan_only_no_source`, `all_three`, `scan_and_prove_*`.
- Uses `lemma` instead of `theorem` for concrete computation results — Isabelle convention: `theorem` for general properties, `lemma` for specific instances.

- [ ] **Step 2: Create the CLI wrapper**

This is the only hand-written Scala. Every mapping function is written explicitly — no hand-waving.

```scala
// verification/oracle/src/main/scala/vulture/OracleCLI.scala
package vulture

import com.google.gson.{Gson, GsonBuilder, JsonArray, JsonObject}
import scala.jdk.CollectionConverters._

object OracleCLI {
  private val gson = new GsonBuilder().setPrettyPrinting().create()

  def main(args: Array[String]): Unit = {
    if (args.length < 2) {
      System.err.println(
        """Usage:
          |  pipeline-oracle advance <pipeline-json> <outcome>
          |  pipeline-oracle expand  <expansion-json>
          |  pipeline-oracle simulate <manifest-path>
          |
          |Examples:
          |  pipeline-oracle advance '{"status":"scan_running","stages":["scan","discover","prove"],"index":0}' completed
          |  pipeline-oracle expand  '{"stages":["prove"],"has_source":true}'
          |  pipeline-oracle simulate verification/simulated-target/manifest.json
        """.stripMargin)
      System.exit(1)
    }

    try {
      args(0) match {
        case "advance"  => cmdAdvance(args)
        case "expand"   => cmdExpand(args)
        case "simulate" => cmdSimulate(args)
        case other =>
          System.err.println(s"Unknown command: $other")
          System.exit(1)
      }
    } catch {
      case e: Exception =>
        System.err.println(s"Error: ${e.getMessage}")
        System.exit(1)
    }
  }

  // ---- advance command ----

  private def cmdAdvance(args: Array[String]): Unit = {
    val json = gson.fromJson(args(1), classOf[JsonObject])
    val status = parseStatus(json.get("status").getAsString)
    val stages = parseStageList(json.getAsJsonArray("stages"))
    val index = BigInt(json.get("index").getAsInt)
    val outcome = parseOutcome(args(2))

    val pipeline = Pipeline_Verified.pipeline_ext(status, stages, index)
    val result = Pipeline_Verified.advance(pipeline, outcome)

    val out = new JsonObject()
    out.addProperty("status", statusToString(Pipeline_Verified.p_status(result)))
    out.addProperty("index", Pipeline_Verified.p_index(result).intValue)
    println(gson.toJson(out))
  }

  // ---- expand command ----

  private def cmdExpand(args: Array[String]): Unit = {
    val json = gson.fromJson(args(1), classOf[JsonObject])
    val stages = parseStageList(json.getAsJsonArray("stages"))
    val hasSource = json.get("has_source").getAsBoolean

    val result = Stage_Expansion_Verified.expand_stages(stages, hasSource)

    val out = new JsonObject()
    val arr = new JsonArray()
    result.foreach(s => arr.add(stageToString(s)))
    out.add("stages", arr)
    println(gson.toJson(out))
  }

  // ---- simulate command (see Task V7) ----

  private def cmdSimulate(args: Array[String]): Unit = {
    val trace = PipelineSimulationOracle.generateTrace(args(1))
    println(PipelineSimulationOracle.toJson(trace))
  }

  // ---- Mapping functions ----
  // These are the highest-risk code in the oracle.
  // A wrong mapping here invalidates all conformance tests.
  // Every case is explicit — no default/wildcard.

  private def parseStatus(s: String): Pipeline_Verified.pipeline_status = s match {
    case "pending"           => Pipeline_Verified.Pending()
    case "scan_running"      => Pipeline_Verified.ScanRunning()
    case "discover_running"  => Pipeline_Verified.DiscoverRunning()
    case "prove_running"     => Pipeline_Verified.ProveRunning()
    case "completed"         => Pipeline_Verified.Completed()
    case "failed"            => Pipeline_Verified.Failed()
    case other => throw new IllegalArgumentException(s"Unknown status: $other")
  }

  private def statusToString(s: Pipeline_Verified.pipeline_status): String = s match {
    case Pipeline_Verified.Pending()          => "pending"
    case Pipeline_Verified.ScanRunning()      => "scan_running"
    case Pipeline_Verified.DiscoverRunning()  => "discover_running"
    case Pipeline_Verified.ProveRunning()     => "prove_running"
    case Pipeline_Verified.Completed()        => "completed"
    case Pipeline_Verified.Failed()           => "failed"
  }

  private def parseStage(s: String): Stage_Expansion_Verified.stage = s match {
    case "scan"     => Stage_Expansion_Verified.Scan()
    case "discover" => Stage_Expansion_Verified.Discover()
    case "prove"    => Stage_Expansion_Verified.Prove()
    case other => throw new IllegalArgumentException(s"Unknown stage: $other")
  }

  private def stageToString(s: Stage_Expansion_Verified.stage): String = s match {
    case Stage_Expansion_Verified.Scan()     => "scan"
    case Stage_Expansion_Verified.Discover() => "discover"
    case Stage_Expansion_Verified.Prove()    => "prove"
  }

  private def parseStageList(arr: JsonArray): List[Stage_Expansion_Verified.stage] =
    arr.asScala.map(e => parseStage(e.getAsString)).toList

  private def parseOutcome(s: String): Pipeline_Verified.audit_outcome = s match {
    case "completed" => Pipeline_Verified.AuditCompleted()
    case "failed"    => Pipeline_Verified.AuditFailed()
    case other => throw new IllegalArgumentException(s"Unknown outcome: $other")
  }
}
```

Notes on Isabelle extraction naming:
- Isabelle extracts `datatype` constructors as case classes with `()`: e.g. `Pending()`, not `Pending`.
- Record fields become accessor functions: `Pipeline_Verified.p_status(result)`, not `result.p_status`.
- Record constructors become `pipeline_ext(...)` (the `_ext` suffix is Isabelle's convention for extensible records).
- `nat` extracts to `BigInt`. The CLI uses `.intValue` for JSON serialization.
- The two theories extract to separate Scala objects (`Pipeline_Verified`, `Stage_Expansion_Verified`) because they declare `stage` independently. The CLI maps between them via string representation.

**IMPORTANT:** The exact Scala API generated by Isabelle varies by version. After running extraction (Task V4 Step 4), inspect the generated `.scala` files and adjust the CLI wrapper if names differ. The mapping functions above are based on Isabelle 2024's code generator conventions.

package vulture

import com.google.gson._
import java.io.FileReader
import scala.jdk.CollectionConverters._

object PipelineSimulationOracle {

  case class StageExpectation(
    stage: String,
    statusBefore: String,
    statusAfter: String,
    minFindings: Int,
    mustFindVulns: List[String],
    mustFindEndpointsMin: Int,
    mustNotFindVulns: List[String]
  )

  case class ExpectedTrace(stages: List[StageExpectation])

  private val gson = new GsonBuilder().setPrettyPrinting().create()

  def generateTrace(manifestPath: String): ExpectedTrace = {
    val root = gson.fromJson(new FileReader(manifestPath), classOf[JsonObject])
    val exp = root.getAsJsonObject("pipeline_expectations")
    val vulns = root.getAsJsonObject("vulnerabilities")

    val mustFindIds = jsonArrayToStrings(vulns.getAsJsonArray("must_find"), "id")
    val mustNotFindIds = jsonArrayToStrings(vulns.getAsJsonArray("must_not_find"), "id")

    val scanExp = exp.getAsJsonObject("scan_to_discover")
    val discoverExp = exp.getAsJsonObject("discover_to_prove")
    val proveExp = exp.getAsJsonObject("prove_completion")

    // Use PROVEN state machine for transitions
    val stages = List(
      Stage_Expansion_Verified.Scan(),
      Stage_Expansion_Verified.Discover(),
      Stage_Expansion_Verified.Prove()
    )
    // Map stage types between the two extracted modules
    val pipelineStages = List(
      Pipeline_Verified.Scan(),
      Pipeline_Verified.Discover(),
      Pipeline_Verified.Prove()
    )
    var pipeline = Pipeline_Verified.pipeline_ext(
      Pipeline_Verified.ScanRunning(), pipelineStages, BigInt(0)
    )

    val traceStages = List(
      {
        val before = OracleCLI.statusToString(Pipeline_Verified.p_status(pipeline))
        pipeline = Pipeline_Verified.advance(pipeline, Pipeline_Verified.AuditCompleted())
        StageExpectation("scan", before, OracleCLI.statusToString(Pipeline_Verified.p_status(pipeline)),
          if (scanExp.get("scan_must_produce_findings").getAsBoolean) 1 else 0,
          mustFindIds, 0, mustNotFindIds)
      },
      {
        val before = OracleCLI.statusToString(Pipeline_Verified.p_status(pipeline))
        pipeline = Pipeline_Verified.advance(pipeline, Pipeline_Verified.AuditCompleted())
        StageExpectation("discover", before, OracleCLI.statusToString(Pipeline_Verified.p_status(pipeline)),
          0, List(), discoverExp.get("discover_must_find_endpoints_min").getAsInt, List())
      },
      {
        val before = OracleCLI.statusToString(Pipeline_Verified.p_status(pipeline))
        pipeline = Pipeline_Verified.advance(pipeline, Pipeline_Verified.AuditCompleted())
        StageExpectation("prove", before, OracleCLI.statusToString(Pipeline_Verified.p_status(pipeline)),
          proveExp.get("prove_must_verify_min").getAsInt, List(), 0, List())
      }
    )

    ExpectedTrace(traceStages)
  }

  def toJson(trace: ExpectedTrace): String = gson.toJson(trace)

  private def jsonArrayToStrings(arr: JsonArray, field: String): List[String] =
    arr.asScala.map(_.getAsJsonObject.get(field).getAsString).toList
}

## Phase 2: Agent-vs-Oracle Verification

**Architecture:**

```
┌──────────────────────┐     ┌────────────────────┐     ┌──────────────────┐
│   Oracle (pure)      │     │   Real Agents       │     │ Simulated Target │
│                      │     │                     │     │                  │
│   Reads manifest     │     │   discover agent    │     │  /.env (exposed) │
│   + proven state     │     │   (Python FastAPI)  │────→│  /openapi.json   │
│   machine            │     │                     │     │  /api/users      │
│                      │     │   Runs 25 plugins   │     │  No HSTS         │
│   Produces:          │     │   against live URL   │     │  X-Powered-By    │
│   expected_findings  │     │                     │     │                  │
│   expected_endpoints │     │   Produces:          │     └──────────────────┘
│   expected_statuses  │     │   actual_findings   │
└──────────┬───────────┘     │   actual_endpoints  │
           │                 └──────────┬──────────┘
           │                            │
           └────────────┬───────────────┘
                        ▼
                ┌───────────────┐
                │  Testbench    │
                │               │
                │  Compare      │
                │  oracle vs    │
                │  agent output │
                │               │
                │  PASS/FAIL    │
                └───────────────┘
```

**Key principle:** The oracle never does I/O. It reads the manifest and produces the deterministic "correct answer." The agents do the real work. The testbench compares. This is the hardware verification pattern: reference model (pure) → implementation (real) → testbench (comparator).

// verification/oracle/src/main/scala/ExpectedResults.scala

import com.google.gson._
import java.io.FileReader
import scala.jdk.CollectionConverters._

/**
 * Reads a manifest and produces a complete expected-results document.
 * No I/O to the target. Pure manifest → expected output transformation.
 *
 * The expected results combine:
 *   1. PROVEN state transitions (from Isabelle-extracted advance function)
 *   2. DETERMINISTIC finding expectations (from manifest ground truth)
 */
object ExpectedResults {
  private val gson = new GsonBuilder().setPrettyPrinting().create()

  case class StageExpected(
    stage: String,
    statusBefore: String,
    statusAfter: String,
    mustFindVulns: List[VulnExpectation],
    mustNotFindVulns: List[FalsePositiveGuard],
    minEndpoints: Int,
    minFindings: Int
  )

  case class VulnExpectation(
    id: String,
    titlePattern: String,
    severityMin: String,
    category: String
  )

  case class FalsePositiveGuard(
    id: String,
    titlePattern: String
  )

  case class FullExpectation(
    targetUrl: String,
    stages: List[StageExpected],
    expectedEndpoints: List[String],
    expectedTechnologies: List[String],
    totalMustFind: Int,
    totalMustNotFind: Int
  )

  def generate(manifestPath: String): FullExpectation = {
    val root = gson.fromJson(new FileReader(manifestPath), classOf[JsonObject])
    val vulns = root.getAsJsonObject("vulnerabilities")
    val endpoints = root.getAsJsonObject("endpoints")
    val tech = root.getAsJsonObject("technologies")
    val exp = root.getAsJsonObject("pipeline_expectations")

    // Parse manifest
    val mustFind = parseVulns(vulns.getAsJsonArray("must_find"))
    val mustNotFind = parseFalsePositives(vulns.getAsJsonArray("must_not_find"))
    val expectedEPs = jsonArrayToStrings(endpoints.getAsJsonArray("expected_discovered"))
    val expectedTech = jsonArrayToStrings(tech.getAsJsonArray("expected_patterns"))

    // Use PROVEN state machine for transitions
    val pipelineStages = List(
      Pipeline_Verified.Scan(), Pipeline_Verified.Discover(), Pipeline_Verified.Prove()
    )
    var pipeline = Pipeline_Verified.pipeline_exta[Unit](
      Pipeline_Verified.ScanRunning(), pipelineStages, Pipeline_Verified.zero_nat(), ()
    )

    val scanExp = exp.getAsJsonObject("scan_to_discover")
    val discoverExp = exp.getAsJsonObject("discover_to_prove")
    val proveExp = exp.getAsJsonObject("prove_completion")

    val stages = List(
      {
        val before = OracleCLI.statusToString(Pipeline_Verified.p_status(pipeline))
        pipeline = Pipeline_Verified.advance(pipeline, Pipeline_Verified.AuditCompleted())
        StageExpected(
          stage = "scan",
          statusBefore = before,
          statusAfter = OracleCLI.statusToString(Pipeline_Verified.p_status(pipeline)),
          mustFindVulns = mustFind,
          mustNotFindVulns = mustNotFind,
          minEndpoints = 0,
          minFindings = if (scanExp.get("scan_must_produce_findings").getAsBoolean) 1 else 0
        )
      },
      {
        val before = OracleCLI.statusToString(Pipeline_Verified.p_status(pipeline))
        pipeline = Pipeline_Verified.advance(pipeline, Pipeline_Verified.AuditCompleted())
        StageExpected(
          stage = "discover",
          statusBefore = before,
          statusAfter = OracleCLI.statusToString(Pipeline_Verified.p_status(pipeline)),
          mustFindVulns = List(),
          mustNotFindVulns = List(),
          minEndpoints = discoverExp.get("discover_must_find_endpoints_min").getAsInt,
          minFindings = 0
        )
      },
      {
        val before = OracleCLI.statusToString(Pipeline_Verified.p_status(pipeline))
        pipeline = Pipeline_Verified.advance(pipeline, Pipeline_Verified.AuditCompleted())
        StageExpected(
          stage = "prove",
          statusBefore = before,
          statusAfter = OracleCLI.statusToString(Pipeline_Verified.p_status(pipeline)),
          mustFindVulns = List(),
          mustNotFindVulns = List(),
          minEndpoints = 0,
          minFindings = proveExp.get("prove_must_verify_min").getAsInt
        )
      }
    )

    FullExpectation(
      targetUrl = root.get("target_url").getAsString,
      stages = stages,
      expectedEndpoints = expectedEPs,
      expectedTechnologies = expectedTech,
      totalMustFind = mustFind.size,
      totalMustNotFind = mustNotFind.size
    )
  }

  def toJson(exp: FullExpectation): String = gson.toJson(exp)

  private def parseVulns(arr: JsonArray): List[VulnExpectation] =
    arr.asScala.map { e =>
      val o = e.getAsJsonObject
      VulnExpectation(
        o.get("id").getAsString,
        o.get("title_pattern").getAsString,
        o.get("severity_min").getAsString,
        if (o.has("category")) o.get("category").getAsString else ""
      )
    }.toList

  private def parseFalsePositives(arr: JsonArray): List[FalsePositiveGuard] =
    arr.asScala.map { e =>
      val o = e.getAsJsonObject
      FalsePositiveGuard(o.get("id").getAsString, o.get("title_pattern").getAsString)
    }.toList

  private def jsonArrayToStrings(arr: JsonArray): List[String] =
    arr.asScala.map(_.getAsString).toList
}

// verification/conformance/agent_harness.go
package conformance

import (
	"bufio"
	"encoding/json"
	"fmt"
	"net/http"
	"os"
	"os/exec"
	"strings"
	"time"
)

// AgentResult holds the structured output from running an agent.
type AgentResult struct {
	Findings  []AgentFinding
	Endpoints []string // from discover_result event
	Status    string   // "completed" or "failed"
}

type AgentFinding struct {
	Title    string `json:"title"`
	Severity string `json:"severity"`
	Category string `json:"category"`
}

// AgentProcess manages a Python agent subprocess.
type AgentProcess struct {
	cmd  *exec.Cmd
	port string
}

// StartAgent launches a Python agent via uvicorn on the given port.
// agentDir is the path to the agent directory (e.g., "../../../agents/discover").
// module is the uvicorn module (e.g., "discover_agent.main:app").
func StartAgent(agentDir, module, port string) (*AgentProcess, error) {
	cmd := exec.Command(
		"python", "-m", "uvicorn", module,
		"--host", "127.0.0.1", "--port", port,
		"--log-level", "warning",
	)
	cmd.Dir = agentDir
	cmd.Env = append(os.Environ(),
		"VULTURE_AGENT_PORT="+port,
		"VULTURE_USE_LLM=false",
	)
	cmd.Stdout = os.Stderr // agent logs go to test stderr
	cmd.Stderr = os.Stderr

	if err := cmd.Start(); err != nil {
		return nil, fmt.Errorf("start agent: %w", err)
	}

	ap := &AgentProcess{cmd: cmd, port: port}
	if err := ap.waitHealthy(10 * time.Second); err != nil {
		ap.Stop()
		return nil, err
	}
	return ap, nil
}

func (ap *AgentProcess) waitHealthy(timeout time.Duration) error {
	deadline := time.Now().Add(timeout)
	url := fmt.Sprintf("http://127.0.0.1:%s/health", ap.port)
	for time.Now().Before(deadline) {
		resp, err := http.Get(url)
		if err == nil && resp.StatusCode == 200 {
			resp.Body.Close()
			return nil
		}
		time.Sleep(200 * time.Millisecond)
	}
	return fmt.Errorf("agent not healthy after %v on port %s", timeout, ap.port)
}

func (ap *AgentProcess) Stop() {
	if ap.cmd != nil && ap.cmd.Process != nil {
		ap.cmd.Process.Kill()
		ap.cmd.Wait()
	}
}

// RunAgent calls POST /run on the agent and collects SSE events.
func RunAgent(port string, request map[string]interface{}) (*AgentResult, error) {
	reqJSON, _ := json.Marshal(request)
	url := fmt.Sprintf("http://127.0.0.1:%s/run", port)

	resp, err := http.Post(url, "application/json", strings.NewReader(string(reqJSON)))
	if err != nil {
		return nil, fmt.Errorf("POST /run: %w", err)
	}
	defer resp.Body.Close()

	result := &AgentResult{}
	scanner := bufio.NewScanner(resp.Body)
	var eventType string

	for scanner.Scan() {
		line := scanner.Text()
		if line == "" {
			eventType = ""
			continue
		}
		if strings.HasPrefix(line, "event: ") {
			eventType = strings.TrimPrefix(line, "event: ")
			continue
		}
		if strings.HasPrefix(line, "data: ") && eventType != "" {
			data := line[6:]
			parseAgentEvent(eventType, data, result)
		}
	}
	return result, nil
}

func parseAgentEvent(eventType, data string, result *AgentResult) {
	switch eventType {
	case "finding":
		var f AgentFinding
		if json.Unmarshal([]byte(data), &f) == nil && f.Title != "" {
			result.Findings = append(result.Findings, f)
		}
	case "discover_result":
		var dr struct {
			URLCount int `json:"url_count"`
			APICount int `json:"api_count"`
		}
		if json.Unmarshal([]byte(data), &dr) == nil {
			// Store counts as synthetic endpoint entries for counting
			for i := 0; i < dr.APICount; i++ {
				result.Endpoints = append(result.Endpoints, fmt.Sprintf("api_endpoint_%d", i))
			}
		}
	case "agent_end":
		result.Status = "completed"
	}
}
